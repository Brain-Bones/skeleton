<Process>
    <ProcessStep step="1">
        ## Create a Project
        Start by creating a new project using `create-svelte-app`:
        ```console
        npm create svelte@latest my-skeleton-app
        cd my-skeleton-app
        npm install
        npm run dev -- --open
        ```
    </ProcessStep>
    <ProcessStep step="2">
        ## Install Skeleton
        Install the Skeleton core and Skeleton Svelte packages.
        ```console
        npm i -D @skeletonlabs/skeleton @skeletonlabs/skeleton-svelte
        ```
    </ProcessStep>
    <ProcessStep step="3">
        ## Install Tailwind
        Install Tailwind using [Svelte-Add](https://github.com/svelte-add/tailwindcss) to automate the process.
        ```console
        npx svelte-add@latest tailwindcss
        npm install
        ```
    </ProcessStep>
    <ProcessStep step="4">
        ## Configure Tailwind
        Open `tailwind.config.js`, then implement the highlighted snippets below.
```js {2, 3, 9, 15-17}
import { join } from 'path';
import { skeleton } from '@skeletonlabs/skeleton';
import * as themes from "@skeletonlabs/skeleton/themes";

/** @type {import('tailwindcss').Config} */
export default {
    content: [
        './src/**/*.{html,js,svelte,ts}',
        join(require.resolve('@skeletonlabs/skeleton-svelte'), '../**/*.{html,js,svelte,ts}')
    ],
    theme: {
        extend: {},
    },
    plugins: [
        skeleton({
            themes: [ themes.cerberus, themes.rose ]
        })
    ]
}
```
    </ProcessStep>
    <ProcessStep step="5">
    ## Set Active Theme
    Open `app.html`, then set the `data-theme` attribute on the body to define the active theme.
```html
<body data-theme="cerberus">...</body>
```
    </ProcessStep>
    <ProcessStep step="check">
        ## Ready
        You're now ready to begin using Skeleton.
    </ProcessStep>
</Process>
